CREATE TABLE PUBLISHER(
    SAP_ID VARCHAR(20) PRIMARY KEY NOT NULL,
    DEPARTMENT VARCHAR(50),
    FIRST_NAME VARCHAR(50),
    MIDDLE_NAME VARCHAR(50),
    LAST_NAME VARCHAR(50),
    PHONE_NUMBER INTEGER(10),
    ROLE VARCHAR(20),
    DATE_OF_JOINING DATE,
    EMAIL VARCHAR(50)
);

CREATE TABLE PAPER(
    PAPER_ID INT PRIMARY KEY NOT NULL AUTO_INCREMENT,
    TITLE VARCHAR(100),
    DOI VARCHAR(100),
    PDF VARCHAR(200)
);

CREATE TABLE ISSUE(
    ISSUEP_ID INT PRIMARY KEY NOT NULL AUTO_INCREMENT,
    PUB_ID VARCHAR(20) REFERENCES PUBLISHER(SAP_ID),
    CATEGORY VARCHAR(20),
    DESCP VARCHAR(150),
    RESPONSE VARCHAR(60),
    ISSUE_STATUS VARCHAR(20),
    TIME_S DATETIME
 );

CREATE TABLE REFERENCE (
    ISSN VARCHAR(12) PRIMARY KEY NOT NULL,
    ISSUE VARCHAR(50),
    MONTH VARCHAR(10),
    PUB_YEAR INTEGER(4),
    PGNO INTEGER(5),
    VOL INTEGER,
    LVL VARCHAR(15),
    ISSN_TYPE VARCHAR(10),
    PUB_TYPE VARCHAR(15),
    NAME VARCHAR(50)
);

CREATE TABLE PAPER_HAS_REFERENCES (
    PAPER_ID INT REFERENCES PAPER(PAPER_ID),
    ISSN VARCHAR(12) REFERENCES REFERENCE(ISSN),
    PRIMARY KEY(PAPER_ID,ISSN)
);

CREATE TABLE AREA_OF_INTEREST (
    SAP_ID VARCHAR(20) REFERENCES PUBLISHER(SAP_ID),
    INTEREST VARCHAR(70),
    PRIMARY KEY(SAP_ID,INTEREST)
);

CREATE TABLE PUB_DETAILS(
    PUBD_ID INT PRIMARY KEY NOT NULL AUTO_INCREMENT,
    SCOPUS_ID VARCHAR(20),
    PUBLON_ID VARCHAR(20),
    H_INDEX INT,
    I_INDEX INT,
    ORCHID VARCHAR(20)
);

CREATE TABLE AUTHORS(
	PAPER_ID INT REFERENCES PAPER(PAPER_ID),
    F_NAME VARCHAR(15),
    M_NAME VARCHAR(15),
    L_NAME VARCHAR(15),
    PRIMARY KEY (PAPER_ID,F_NAME)
);

CREATE TABLE PUBLISHER_HAS_DETAILS(
	SAP_ID VARCHAR(20) REFERENCES PUBLISHER(SAP_ID),
    PUBD_ID INT REFERENCES PUBLISHER_DETAILS(PUBD_ID),
    PRIMARY KEY (SAP_ID,PUBD_ID)
);

CREATE TABLE PUBLISHES(
	SAP_ID VARCHAR(20) REFERENCES PUBLISHER(SAP_ID),
    PAPER_ID INT REFERENCES PAPER(PAPER_ID),
    PRIMARY KEY(SAP_ID,PAPER_ID)
);
